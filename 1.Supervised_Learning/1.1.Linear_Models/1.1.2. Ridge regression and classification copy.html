import matplotlib.pyplot as plt
from sklearn import linear_model

# X is the 10x10 Hilbert matrix

y = np.ones(10)

n_alphas = 200
alphas = np.logspace(-10, -2, n_alphas)

coefs = []

    ridge = linear_model.Ridge(alpha = a, fit_intercept = False)
    ridge.fit(X,y)
